cd ~/Downloads/project/APPLY/openclaw

# 1. Set gateway mode
pnpm openclaw config set gateway.mode local

# 2. (Optional) Silence memory warning
pnpm openclaw config set agents.defaults.memorySearch.enabled false

# 3. Start gateway
pnpm openclaw gateway run

# 4. (Wait ~5 seconds, then in a NEW Git Bash tab)
pnpm openclaw gateway status

# 5. Get the tokenized dashboard URL
pnpm openclaw dashboard --no-open

# 6. Copy the full URL into your browser

# 7. Add the cron job
pnpm openclaw cron add --name "daily-job-apply" --cron "0 9 * * 1-5" --timeout-seconds 3600 --message "Execute job auto-apply per standing orders. Activate the job_auto_apply skill. Process all 13 target companies. Log results and report summary."
